1-   git init                     ( empezar repo )
2-   git add .                    ( a√±adir todos los archivos )
3-   git reset .                  ( quitar todos los archivos )
4-   git commit -m  "la m es para comentarios"
5-   git checkout -- .            ( reconstruir archivos al ultimo commit )
6-   git log

8-   git checkout -b nombre_rama  ( crear rama )

12-  git branch -d nombre_rama   ( borrar rama )
13.  git push                      
14.  git commit -am "agregado"    ( git commit + git add )
